## Code Example: React Component

```javascript
const Welcome = ({ name }) => {
    const [greeting, setGreeting] = useState('Hello');
    
    return (
        <div className="welcome-component">
            <h1>{greeting}, {name}!</h1>
            <button onClick={() => setGreeting('Hi')}>
                Change Greeting
            </button>
        </div>
    );
};
```
